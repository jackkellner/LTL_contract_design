## COMMENTS
##
## Contract 1 represents the Waiter, contract 2 represents the Customer
##
## Variable Definitions:
## request - the customer has requested service
## service - the waiter is servicing the customer

CONTRACT:
	NAME:
    contract1
  VARIABLES:
    request := FALSE
    service := FALSE
	ASSUMPTIONS:
		TRUE
	GUARANTEES:
		G(!request -> X !service)
		G(request -> X service)

CONTRACT:
	NAME: 
	  contract2
  VARIABLES:
    request := FALSE
    service := FALSE
	ASSUMPTIONS:
		TRUE
	GUARANTEES:
		(F request)
		G((request & !service) -> X request)
		G(service -> X !request)

CHECKS:
	compatibility(contract1, contract2)
	consistency(contract1, contract2)
